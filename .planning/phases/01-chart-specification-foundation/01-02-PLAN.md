```typescript
import { tool } from "ai";
import { z } from "zod";
import Papa from "papaparse";

// ─── Types ──────────────────────────────────────────────────────────────────

/** Validated chart specification produced by the AI tool call. */
export interface ChartSpec {
  type: "bar" | "line" | "pie" | "scatter";
  x_column: string;
  y_column: string;
  title: string;
  insight_caption?: string;
  data: Record<string, unknown>[];
}

// ─── Tool builder ────────────────────────────────────────────────────────────

/**
 * Build a Vercel AI SDK tool() object for chart_spec with a dynamic Zod enum
 * constructed from the session's actual column names. Must be called inside
 * the POST handler (not at module level) because it depends on runtime columns.
 *
 * @throws Error if columns is empty — caller must return 400 before using this
 */
export function buildChartSpecTool(columns: string[]) {
  if (columns.length === 0) {
    throw new Error("Session has no columns");
  }

  const columnEnum = z.enum(columns as [string, ...string[]]);

  return tool({
    description:
      "Generate a chart specification to visualize CSV data. " +
      "Call this when the question implies comparison, trend, distribution, or " +
      "relationship between values. Do NOT call for simple single-value lookups.",
    parameters: z.object({
      type: z
        .enum(["bar", "line", "pie", "scatter"])
        .describe("Chart type: bar=categories, line=time series, pie=parts of whole (≤6 categories), scatter=two numeric variables"),
      x_column: columnEnum.describe("Exact column name for the x-axis / label axis"),
      y_column: columnEnum.describe("Exact column name for the y-axis / value axis"),
      title: z.string().describe("A concise, descriptive chart title"),
      insight_caption: z
        .string()
        .optional()
        .describe("One-sentence insight summarising the key takeaway from the chart"),
      data: z
        .array(z.record(z.string(), z.unknown()))
        .describe("Up to 200 rows of CSV data, all columns included"),
    }),
    // No execute — this is a client-side tool.
    // The AI generates the tool call; Phase 2 (ChartBlock) renders it.
  });
}

// ─── Row extraction ──────────────────────────────────────────────────────────

const MAX_ROWS = 200;
const CHUNKS_NEEDED = 4; // 4 chunks × 50 rows/chunk = 200 rows

/**
 * Parse up to 200 rows from csv_chunks chunk text.
 * Chunk 0 contains the header row; subsequent chunks do not.
 * Uses PapaParse to correctly handle quoted fields containing commas.
 *
 * @param chunks - Array of { chunkText: string } in chunkIndex order (ascending)
 * @returns Array of row objects (up to 200), keys are column names
 */
export function parseRowsFromChunks(
  chunks: { chunkText: string }[]
): Record<string, unknown>[] {
  if (chunks.length === 0) return [];

  // Take only the first CHUNKS_NEEDED chunks (performance guard)
  const relevant = chunks.slice(0, CHUNKS_NEEDED);

  // Reconstruct CSV text: chunk 0 has header, chunks 1+ have data only
  const firstLines = relevant[0].chunkText.split("\n");
  const header = firstLines[0] ?? "";

  const bodyLines = relevant.flatMap((c, i) => {
    const lines = c.chunkText.split("\n");
    return i === 0 ? lines.slice(1) : lines;
  });

  const csvText = [header, ...bodyLines].filter(Boolean).join("\n");

  const parsed = Papa.parse<Record<string, unknown>>(csvText, {
    header: true,
    skipEmptyLines: true,
    dynamicTyping: true, // keep numbers as numbers for Recharts numeric axes
  });

  return parsed.data.slice(0, MAX_ROWS);
}
```

- The `as [string, ...string[]]` cast is required for TypeScript to accept z.enum with a runtime array
- `dynamicTyping: true` in PapaParse keeps Recharts' numeric axis data as numbers

import { describe, it, expect } from "vitest";
import { buildChartSpecTool, parseRowsFromChunks } from "../chart-spec";

// ─── buildChartSpecTool ──────────────────────────────────────────────────────

describe("buildChartSpecTool", () => {
  const columns = ["Name", "Value", "Category"];

  it("throws when columns array is empty", () => {
    expect(() => buildChartSpecTool([])).toThrow("Session has no columns");
  });

  it("returns a tool object with a parameters schema", () => {
    const t = buildChartSpecTool(columns);
    expect(t).toHaveProperty("parameters");
  });

  describe("type field", () => {
    it("accepts valid chart types", () => {
      const schema = buildChartSpecTool(columns).parameters;
      for (const type of ["bar", "line", "pie", "scatter"]) {
        const result = schema.safeParse({
          type,
          x_column: "Name",
          y_column: "Value",
          title: "Test Chart",
          data: [],
        });
        expect(result.success, `type "${type}" should be valid`).toBe(true);
      }
    });

    it("rejects invalid chart types", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "histogram",
        x_column: "Name",
        y_column: "Value",
        title: "Test",
        data: [],
      });
      expect(result.success).toBe(false);
    });
  });

  describe("column validation (VALID-01, VALID-04)", () => {
    it("accepts x_column and y_column that are in the columns list", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "bar",
        x_column: "Name",
        y_column: "Value",
        title: "Valid",
        data: [],
      });
      expect(result.success).toBe(true);
    });

    it("rejects x_column not in the columns list", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "bar",
        x_column: "NonExistent",
        y_column: "Value",
        title: "Invalid",
        data: [],
      });
      expect(result.success).toBe(false);
    });

    it("rejects y_column not in the columns list", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "bar",
        x_column: "Name",
        y_column: "HallucinatedColumn",
        title: "Invalid",
        data: [],
      });
      expect(result.success).toBe(false);
    });
  });

  describe("optional fields", () => {
    it("accepts missing insight_caption", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "scatter",
        x_column: "Name",
        y_column: "Value",
        title: "No Caption",
        data: [],
      });
      expect(result.success).toBe(true);
    });

    it("accepts insight_caption when provided", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "line",
        x_column: "Name",
        y_column: "Value",
        title: "With Caption",
        insight_caption: "Values peak in Q3.",
        data: [{ Name: "Q1", Value: 10 }],
      });
      expect(result.success).toBe(true);
    });

    it("rejects missing title", () => {
      const schema = buildChartSpecTool(columns).parameters;
      const result = schema.safeParse({
        type: "bar",
        x_column: "Name",
        y_column: "Value",
        data: [],
      });
      expect(result.success).toBe(false);
    });
  });
});

// ─── parseRowsFromChunks ─────────────────────────────────────────────────────

describe("parseRowsFromChunks", () => {
  it("returns empty array for empty input", () => {
    expect(parseRowsFromChunks([])).toEqual([]);
  });

  it("parses rows from a single chunk with header", () => {
    const chunks = [{ chunkText: "A,B\n1,2\n3,4" }];
    const rows = parseRowsFromChunks(chunks);
    expect(rows).toHaveLength(2);
    expect(rows[0]).toMatchObject({ A: 1, B: 2 }); // dynamicTyping: numbers
    expect(rows[1]).toMatchObject({ A: 3, B: 4 });
  });

  it("caps rows at 200 (VALID-03)", () => {
    // Build 5 chunks of 50 rows each = 250 rows total, expect 200
    const headerRow = "X,Y";
    const dataRows = Array.from({ length: 50 }, (_, i) => `${i},${i * 2}`).join("\n");
    const chunk0 = { chunkText: `${headerRow}\n${dataRows}` };
    const chunkN = { chunkText: dataRows };

    const chunks = [chunk0, chunkN, chunkN, chunkN, chunkN]; // 5 chunks × 50 = 250
    const rows = parseRowsFromChunks(chunks);
    expect(rows).toHaveLength(200);
  });

  it("handles quoted fields containing commas", () => {
    const chunks = [{ chunkText: `Name,Value\n"Smith, John",42\n"Doe, Jane",99` }];
    const rows = parseRowsFromChunks(chunks);
    expect(rows).toHaveLength(2);
    expect(rows[0]).toMatchObject({ Name: "Smith, John", Value: 42 });
  });

  it("skips empty lines", () => {
    const chunks = [{ chunkText: "A,B\n1,2\n\n3,4\n" }];
    const rows = parseRowsFromChunks(chunks);
    expect(rows).toHaveLength(2);
  });
});

| Boundary | Description |
|----------|-------------|
| POST handler → tool() schema | Column names from DB used to construct Zod enum; AI-supplied x_column/y_column validated against this enum |
| AI tool call arguments → application | Zod parameters schema validates all fields before any application code sees them |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | buildChartSpecTool column enum | mitigate | Columns sourced from `csv_sessions.columns` (DB, server-side) — not from client request body per D-07. Dynamic enum enforces valid column names; any other value fails Zod validation before reaching route logic. |
| T-02-02 | Tampering | chart type enum | mitigate | `z.enum(["bar","line","pie","scatter"])` — AI cannot inject unsupported chart types. Validated by Zod before Plan 03 route code runs. |
| T-02-03 | Denial of Service | parseRowsFromChunks large payload | mitigate | Hard cap of 200 rows (`MAX_ROWS = 200`) + `CHUNKS_NEEDED = 4` limits processing to at most 4 chunks. Extra chunks are ignored. |
| T-02-04 | Information Disclosure | insight_caption optional field | accept | Caption is AI-generated text from the session's own data. User already has access to this data. No cross-user disclosure risk. |
</threat_model>
