```typescript
const sessionRow = await db
  .select({ columns: sessions.columns, userId: sessions.userId })
  .from(sessions)
  .where(eq(sessions.id, sessionId))
  .limit(1);

if (!sessionRow[0]) {
  trace.update({ output: { ok: false, error: "session_not_found" } });
  langfuse.flushAsync().catch(() => {});
  return Response.json({ error: "Session not found" }, { status: 404 });
}
if (sessionRow[0].userId !== session.user.id) {
  trace.update({ output: { ok: false, error: "forbidden" } });
  langfuse.flushAsync().catch(() => {});
  return Response.json({ error: "Forbidden" }, { status: 403 });
}

const columns = sessionRow[0].columns ?? [];
if (columns.length === 0) {
  trace.update({ output: { ok: false, error: "no_columns" } });
  langfuse.flushAsync().catch(() => {});
  return Response.json({ error: "Session has no column data" }, { status: 400 });
}

// ── Row data for chart_spec (200-row cap — VALID-03) ──────────────────────
const rawChunks = await db
  .select({ chunkText: csvChunks.chunkText })
  .from(csvChunks)
  .where(eq(csvChunks.sessionId, sessionId))
  .orderBy(asc(csvChunks.chunkIndex))
  .limit(4); // 4 chunks × 50 rows = 200 rows max

const rowData = parseRowsFromChunks(rawChunks);
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → POST /api/chat | sessionId is untrusted client input; must be verified against session.user.id |
| POST handler → streamText tools | AI-generated tool call arguments pass through Zod parameters schema before any application code |
| DB columns → Zod enum | Column names from csv_sessions.columns are the authoritative source; never from client body |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | sessionId IDOR in /api/chat | mitigate | Query `sessions.userId` alongside `sessions.columns` and compare to `session.user.id`. Return 403 if mismatch. This is ASVS V4 (Access Control) and was explicitly identified as blocking in RESEARCH.md security domain. Implementation: `sessionRow[0].userId !== session.user.id → 403`. |
| T-03-02 | Tampering | AI tool call column injection | mitigate | buildChartSpecTool constructs a `z.enum([...columns])` from DB-sourced column names. Any AI-hallucinated column name fails Zod validation before reaching application code. Column source is DB (D-07), never client request body. |
| T-03-03 | Denial of Service | Large data payload in chart_spec.data | mitigate | parseRowsFromChunks caps at MAX_ROWS=200 and processes at most CHUNKS_NEEDED=4 chunks. Inline data in tool call is bounded. |
| T-03-04 | Denial of Service | Infinite AI tool call loop | mitigate | `maxSteps: 1` set in streamText call. Prevents AI from repeatedly calling chart_spec in a loop. |
| T-03-05 | Spoofing | columns list from client body | mitigate | Columns fetched from `csv_sessions` (DB) after ownership verification. Request body is never used as column source per D-07. |
| T-03-06 | Information Disclosure | Session data of another user | mitigate | IDOR check (T-03-01) prevents querying another user's session. RAG context (buildRAGContext) is also scoped by sessionId in WHERE clause (existing implementation). |
</threat_model>
